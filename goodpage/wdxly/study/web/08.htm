Step 1 :首先将下面的代码加入HTML的<head>区。
<script>

var imgwidth=400
var imgheight=60

//下面定义了包含2个图片的数组，数组的下标从0开始，你可以增加任意的图片，按照顺序增加即可，注意相应增加数组下标。
var imgname=new Array()
imgname[0]="imggum1.jpg"
imgname[1]="imggum2.jpg"

imgurl[1]="#"

for (i=0;i<=imgname.length-1;i++) {
imgpreload[i]=new Image()
imgpreload[i].src=imgname[i]
}


var pause=2000

//图片的显示速度，数值小速度快；
var speed=20

//下面的参数也会影响图片显示的速度，数值小速度快；代表2个图片的显示间隔。
var step=10

//不要修改下面的代码；
var i_loop=0
var i_image=0

function stretchimage() {
if (i_loop<=imgwidth) {
if (document.all) {
imgcontainer.innerHTML="<a href='"+imgurl[i_image]+"' target='_blank'><img width='"+i_loop+"' height='"+imgheight+"' src='"+imgname[i_image]+"' border='0'></a>"
}
i_loop=i_loop+step
var timer=setTimeout("stretchimage()",speed)
}
else {
i_loop=imgwidth
clearTimeout(timer)
imgcontainer.innerHTML="<a href='"+imgurl[i_image]+"' target='_blank'><img src='"+imgname[i_image]+"' border='0'></a>"

}
}

function shrinkimage() {
if (i_loop>=0) {
if (document.all) {
imgcontainer.innerHTML="<a href='"+imgurl[i_image]+"' target='_blank'><img width='"+i_loop+"' height='"+imgheight+"' src='"+imgname[i_image]+"' border='0'></a>"
}
i_loop=i_loop-step
var timer=setTimeout("shrinkimage()",speed)
}
else {
i_loop=0
clearTimeout(timer)
changeimage()
}
}

function changeimage() {
i_image++
if (i_image>=imgname.length) {i_image=0}

}
stretchimage()
}

Step 2: 将CSS控制加入<head>区
<style>
.containerstyle {
position:absolute;
left:60px;
top:20px;
}
</style>

</head>

<body bgcolor="#FFFFFF"onLoad="stretchimage()">
<span id="imgcontainer"class="containerstyle"></span>
</body>
</html>

最后，在<Body>标签内加上onload事件句柄，在<body>区加上<span>...</span>标签。你可能需要修改图片出现的绝对位置。如上兰色部分。修改直至位置符合你的具体需要。
